[703. 数据流中的第 K 大元素](https://leetcode-cn.com/problems/kth-largest-element-in-a-stream/)

```C++
class KthLargest {//第一次做自己只会用sort，这是别人的正确做法，用堆来排序
public:
    KthLargest(int k, vector<int>& nums) {
        size = k;
        for (auto &i : nums) {
            worker.push(i);
            // 始终保持队列(堆)大小为k
            if (worker.size() > size) worker.pop();
        }
    }

    int add(int val) {
        worker.push(val);
        // 始终保持队列(堆)大小为k
        if (worker.size() > size) worker.pop();
        return worker.top();
    }

private:
    int size;
    // 显示构造成小顶堆
    priority_queue<int, vector<int>, greater<int>> worker;
};
```

public:
    stack<int> st1;
    stack<int> st2;

    MyQueue() {

    }
    
    /** Push element x to the back of queue. */
    void push(int x) {
        st1.push(x);
    }
    
    /** Removes the element from in front of queue and returns that element. */
    int pop() {
        while(!st1.empty()){
            st2.push(st1.top());
            st1.pop();
        }
        int temp=st2.top();
        st2.pop();
        while(!st2.empty()){
            st1.push(st2.top());
            st2.pop();
        }
        return temp;
    }
    
    /** Get the front element. */
    int peek() {
        int temp=this->pop();
        while(!st1.empty()){
            st2.push(st1.top());
            st1.pop();
        }
        st1.push(temp);
        while(!st2.empty()){
            st1.push(st2.top());
            st2.pop();
        }
        return temp;
    }
    
    /** Returns whether the queue is empty. */
    bool empty() {
        return st1.empty()&&st2.empty();
    }
};
